public final class TtlRunnable implements Runnable {

    private final Runnable runnable;
    
    private TtlRunnable(Runnable runnable) {
        this.captured = Transmitter.capture(); // æ„é€ æ—¶æ•è·ä¸Šä¸‹æ–‡
