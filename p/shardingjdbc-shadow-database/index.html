@Service
public class OrderService {

    private OrderMapper orderMapper;
    
    @Autowired
    private OrderItemMapper orderItemMapper;
    
    /**
     * åˆ›å»ºè®¢å• - æ”¯æŒå½±å­åº“è·¯ç”±

    @Transactional
    public Long createOrder(CreateOrderRequest request) {
        // æ„å»ºè®¢å•å¯¹è±¡
        Order order = Order.builder()
                .userId(request.getUserId())
                .totalAmount(request.getTotalAmount())
                .status(OrderStatus.PENDING)
                .createTime(LocalDateTime.now())
                .build();
        
        // æ’å…¥è®¢å• - ä¼šæ ¹æ®userIdè‡ªåŠ¨è·¯ç”±åˆ°å¯¹åº”åº“
        orderMapper.insert(order);

                .map(item -> OrderItem.builder()
                        .orderId(order.getId())
                        .userId(request.getUserId()) // å…³é”®ï¼šä¿æŒuserIdä¸€è‡´
                        .productId(item.getProductId())
                        .quantity(item.getQuantity())
                        .price(item.getPrice())
                        .build())
                .collect(Collectors.toList());
        
        orderItemMapper.batchInsert(orderItems);
        
        return order.getId();
    }
    
    /**
     * æŸ¥è¯¢è®¢å•
     */
    public OrderVO getOrder(Long orderId, Long userId) {
        // æ ¹æ®userIdè·¯ç”±åˆ°å¯¹åº”çš„æ•°æ®åº“

        if (order == null) {
            return null;
        }
        
        List<OrderItem> items = orderItemMapper.selectByOrderId(orderId, userId);
        
        return OrderVO.builder()
                .order(order)
                .items(items)
                .build();
    }
}
